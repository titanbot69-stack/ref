// My Requests description toggles
document.addEventListener('DOMContentLoaded',function(){
	const table=document.getElementById('my-requests-table');
	if(!table) return;
	table.addEventListener('click',function(e){
		const toggle=e.target.closest('.desc-toggle');
		if(!toggle) return;
		e.preventDefault();
		const wrap=toggle.closest('.desc-wrapper');
		if(!wrap) return;
		const shortEl=wrap.querySelector('.desc-short');
		const fullEl=wrap.querySelector('.desc-full');
		const expanded = wrap.getAttribute('data-state') === 'expanded';
		if(expanded){
			fullEl.classList.add('d-none');
			shortEl.classList.remove('d-none');
			wrap.setAttribute('data-state','collapsed');
			toggle.textContent='Read More';
		} else {
			shortEl.classList.add('d-none');
			fullEl.classList.remove('d-none');
			wrap.setAttribute('data-state','expanded');
			toggle.textContent='Read Less';
		}
	});
});
