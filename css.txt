// My Requests description toggles
document.addEventListener('DOMContentLoaded',function(){
	const table=document.getElementById('my-requests-table');
	if(!table) return;
	table.addEventListener('click',function(e){
		const toggle=e.target.closest('.desc-toggle');
		if(!toggle) return;
		e.preventDefault();
		const wrap=toggle.closest('.desc-wrapper');
		if(!wrap) return;
		const expanded = wrap.getAttribute('data-state') === 'expanded';
		wrap.setAttribute('data-state', expanded ? 'collapsed' : 'expanded');
		toggle.textContent = expanded ? 'Read More' : 'Read Less';
		toggle.setAttribute('aria-expanded', (!expanded).toString());
	});
});
