        requestList.addEventListener('click', function (e) {
            const toggleLink = e.target.closest('.toggle-description');
            if (!toggleLink) return;

            const eventDescription = toggleLink.closest('.event-description');
            const shortDescription = eventDescription.querySelector('.short-description');
            const fullDescription = eventDescription.querySelector('.full-description');

            if (shortDescription.style.display === 'none') {
                shortDescription.style.display = '';
                fullDescription.style.display = 'none';
                toggleLink.textContent = 'Read More';
            } else {
                shortDescription.style.display = 'none';
                fullDescription.style.display = '';
                toggleLink.textContent = 'Read Less';
            }

            e.preventDefault();
        });
