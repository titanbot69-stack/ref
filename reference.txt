<script>
    document.addEventListener('DOMContentLoaded', function () {
        const searchInput = document.querySelector('[data-table-id="app-owner-requests-table-container"]');
        const requestList = document.getElementById('app-owner-requests-table-container');
        const requestItems = requestList.querySelectorAll('.request-item');

        searchInput.addEventListener('input', function () {
            const searchTerm = this.value.toLowerCase();

            requestItems.forEach(item => {
                const searchableText = item.getAttribute('data-searchable').toLowerCase();
                if (searchableText.includes(searchTerm)) {
                    item.style.display = '';
                } else {
                    item.style.display = 'none';
                }
            });
        });

        const toggleLinks = document.querySelectorAll('.toggle-description');

        toggleLinks.forEach(link => {
            link.addEventListener('click', function (e) {
                e.preventDefault();
                const parent = this.closest('.event-description');
                const shortDesc = parent.querySelector('.short-description');
                const fullDesc = parent.querySelector('.full-description');

                if (shortDesc.style.display === 'none') {
                    shortDesc.style.display = '';
                    fullDesc.style.display = 'none';
                    this.textContent = 'Read More';
                } else {
                    shortDesc.style.display = 'none';
                    fullDesc.style.display = '';
                    this.textContent = 'Read Less';
                }
            });
        });
    });
</script>
